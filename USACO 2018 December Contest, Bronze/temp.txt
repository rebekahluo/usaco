#include <bits/stdc++.h>
using namespace std;
vector<int> a;
vector<int> b;
int t1, t2;
int c[2000 + 10];
void friday(int t1, vector<int> a, vector<int> b)
{
	for (int i = 0; i < b.size(); i++)
	{
		int nai = b[i];
		vector<int> aa = a;
		vector<int> bb = b;
		bb.erase(bb.begin() + i);
		aa.push_back(nai);
		
		
		c[t1 + nai] = 1;
	}
}

void thursday(int t1, vector<int> a, vector<int> b)
{
	for (int i = 0; i < a.size(); i++)
	{
		int nai = a[i];
		
		vector<int> aa = a;
		vector<int> bb = b;
		aa.erase(aa.begin() + i);
		
		bb.push_back(nai);
		friday(t1 - nai, aa, bb); 
	}
}

void wednsday(int t1, vector<int> a, vector<int> b)
{

	for (int i = 0; i < b.size(); i++)
	{
		int nai = b[i];
		
		vector<int> aa = a;
		vector<int> bb = b;
		bb.erase(bb.begin() + i);
		aa.push_back(nai);
		thursday(t1 + nai, aa, bb);
	}
}

void tuesday(int t1, vector<int> a, vector<int> b)
{
	for (int i = 0; i < a.size(); i++)
	{
		int nai = a[i];
	//	t1 -= a[i];
	//	t2 += a[i];
		
		vector<int> aa = a;
		vector<int> bb = b;
		
		a.erase(a.begin() + i);
		b.push_back(nai);
		
		wednsday(t1 - nai, aa, bb);
	}	
}

int monday()
{
	t1 = 1000;
	t2 = 1000;
	tuesday(t1, a, b);
}
